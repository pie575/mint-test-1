---

每个 Mintlify 站点都需要一个包含其核心配置的 `docs.json` 文件。详情请参阅下方的[属性](#properties)。

<div id="properties">
  ## 属性
</div>

<ResponseField name="name" type="string" required>
  项目名称，用于全局标题。

  示例：`mintlify`
</ResponseField>

<ResponseField name="navigation" type="Navigation[]" required>
  由导航分组及其页面组成的数组

  <Expandable title="导航">
    <ResponseField name="group" type="string">
      分组名称。

      示例：`Settings`
    </ResponseField>

    <ResponseField name="pages" type="string[]">
      作为页面使用的 MDX 文件的相对路径。

      示例：`["customization", "page"]`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="logo" type="string or object">
  Logo 图片路径，或包含单独的“浅色”和“深色”模式 Logo 路径的对象

  <Expandable title="Logo">
    <ResponseField name="light" type="string">
      浅色模式下的 Logo 路径
    </ResponseField>

    <ResponseField name="dark" type="string">
      深色模式下的 Logo 路径
    </ResponseField>

    <ResponseField name="href" type="string" default="/">
      点击 Logo 后跳转到的 URL
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="favicon" type="string">
  favicon 图片路径
</ResponseField>

<ResponseField name="colors" type="Colors">
  全局主题使用的十六进制颜色代码

  <Expandable title="Colors">
    <ResponseField name="primary" type="string" required>
      主色，最常用于高亮内容、章节标题
      以及浅色模式下的强调色
    </ResponseField>

    <ResponseField name="light" type="string">
      深色模式下的主色，最常用于高亮
      内容、章节标题以及深色模式下的强调色
    </ResponseField>

    <ResponseField name="dark" type="string">
      重要按钮使用的主色
    </ResponseField>

    <ResponseField name="background" type="object">
      浅色和深色模式下的背景颜色

      <Expandable title="Object">
        <ResponseField name="light" type="string" required>
          浅色模式下背景的十六进制颜色代码
        </ResponseField>

        <ResponseField name="dark" type="string" required>
          深色模式下背景的十六进制颜色代码
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarLinks" type="TopbarLink[]">
  顶部栏中要显示的链接数组

  <Expandable title="TopbarLink">
    <ResponseField name="name" type="string">
      按钮名称。

      示例：`Contact us`
    </ResponseField>

    <ResponseField name="url" type="string">
      按钮跳转到的 URL。示例：`https://mintlify.com/docs`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topbarCtaButton" type="Call to Action">
  <Expandable title="顶部栏行动号召">
    <ResponseField name="type" type={'"link" or "github"'} default="link">
      `link` 会显示一个按钮；`github` 会显示仓库信息，包括 star 数量。
    </ResponseField>

    <ResponseField name="url" type="string">
      如果是 `link`：按钮跳转到的目标。

      如果是 `github`：用于加载 GitHub 信息的仓库链接。
    </ResponseField>

    <ResponseField name="name" type="string">
      按钮中的文本。仅当 `type` 为 `link` 时需要填写。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="versions" type="string[]">
  版本名称数组。仅当你希望通过导航栏中的下拉菜单显示
  不同版本的文档时才使用此项。
</ResponseField>

<ResponseField name="anchors" type="Anchor[]">
  锚点链接数组，每个锚点链接都包含 `icon`、`color` 和 `url`。

  <Expandable title="锚点链接">
    <ResponseField name="icon" type="string">
      用于显示该锚点链接的 [Font Awesome](https://fontawesome.com/search?q=heart) 图标。

      示例：`comments`
    </ResponseField>

    <ResponseField name="name" type="string">
      锚点链接标签的名称。

      示例：`Community`
    </ResponseField>

    <ResponseField name="url" type="string">
      用于标识哪些页面归入该锚点链接的 URL 起始部分。通常，这是你存放页面的文件夹名称。
    </ResponseField>

    <ResponseField name="color" type="string">
      锚点链接图标背景的十六进制颜色。也可以传入一个包含 `from` 和 `to` 属性的对象来使用渐变色，这两个属性的值都应为十六进制颜色。
    </ResponseField>

    <ResponseField name="version" type="string">
      如果你希望仅在选择了正确的文档版本后才显示某个锚点链接，可使用此项。
    </ResponseField>

    <ResponseField name="isDefaultHidden" type="boolean" default="false">
      如果你希望默认隐藏该锚点链接，直到有人通过直接链接访问其中的文档时才显示，请传入 `true`。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      可选值之一：&quot;brands&quot;、&quot;duotone&quot;、&quot;light&quot;、&quot;sharp-solid&quot;、&quot;solid&quot; 或 &quot;thin&quot;
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="topAnchor" type="Object">
  覆盖最顶部锚点链接的默认配置。

  <Expandable title="Object">
    <ResponseField name="name" default="Documentation" type="string">
      最顶部锚点链接的名称
    </ResponseField>

    <ResponseField name="icon" default="book-open" type="string">
      Font Awesome 图标。
    </ResponseField>

    <ResponseField name="iconType" default="duotone" type="string">
      可选值之一：&quot;brands&quot;、&quot;duotone&quot;、&quot;light&quot;、&quot;sharp-solid&quot;、&quot;solid&quot; 或 &quot;thin&quot;
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="tabs" type="Tabs[]">
  标签页数组。

  <Expandable title="Tabs">
    <ResponseField name="name" type="string">
      标签页标签的名称。
    </ResponseField>

    <ResponseField name="url" type="string">
      用于标识哪些页面归入该标签页的 URL 起始部分。通常，这
      是你存放页面的文件夹名称。
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="api" type="API">
  API 设置相关配置。有关 API 页面的更多信息，请参阅 [API Components](/zh/api-playground/demo)。

  <Expandable title="API">
    <ResponseField name="baseUrl" type="string">
      所有 API 端点的基础 URL。如果 `baseUrl` 是数组，用户可以在多个基础 URL 选项之间切换。
    </ResponseField>

    <ResponseField name="auth" type="Auth">
      <Expandable title="Auth">
        <ResponseField name="method" type="&#x22;bearer&#x22; | &#x22;basic&#x22; | &#x22;key&#x22;">
          适用于所有 API 端点的身份验证策略。
        </ResponseField>

        <ResponseField name="name" type="string">
          API playground 中使用的身份验证参数名称。

          如果 method 为 `basic`，格式应为 `[usernameName]:[passwordName]`
        </ResponseField>

        <ResponseField name="inputPrefix" type="string">
          身份验证输入字段的默认前缀。

          例如，如果 `inputPrefix` 为 `AuthKey`，身份验证字段会预先填入 `AuthKey`。
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="playground" type="Playground">
      API playground 的配置

      <Expandable title="Playground">
        <ResponseField name="mode" default="show" type="&#x22;show&#x22; | &#x22;simple&#x22; | &#x22;hide&#x22;">
          控制 playground 是显示、隐藏，还是仅显示端点而不提供额外的用户交互 (`simple`)

          更多信息请参阅 [playground 指南](/zh/api-playground/demo)
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="maintainOrder" type="boolean">
      启用此标志后，可确保 OpenAPI 页面中的键顺序与 OpenAPI 文件中定义的键顺序一致。

      <Warning>此行为很快将默认启用，届时该字段将被弃用。</Warning>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="openapi" type="string | string[]">
  指向你的 OpenAPI 文件的 URL 或相对路径。也可以是包含
  多个文件的数组。

  示例：

  <CodeGroup>
    ```json Absolute
    "openapi": "https://example.com/openapi.json"
    ```

    ```json Relative
    "openapi": "/openapi.json"
    ```

    ```json Multiple
    "openapi": ["https://example.com/openapi1.json", "/openapi2.json", "/openapi3.json"]
    ```
  </CodeGroup>
</ResponseField>

<ResponseField name="footerSocials" type="FooterSocials">
  一个社交媒体账号对象，其中每个键值对都将平台映射到对应的 URL。

  示例：

  ```json
  {
    "x": "https://x.com/mintlify",
    "website": "https://mintlify.com"
  }
  ```

  <Expandable title="FooterSocials">
    <ResponseField name="[key]" type="string">
      以下值之一：`website`、`facebook`、`x`、`discord`、`slack`、`github`、`linkedin`、`instagram`、`hacker-news`

      示例：`x`
    </ResponseField>

    <ResponseField name="property" type="string">
      社交平台的 URL。

      示例：`https://x.com/mintlify`
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="feedback" type="Feedback">
  配置文档页面上的反馈按钮

  <Expandable title="Feedback">
    <ResponseField name="suggestEdit" type="boolean" default="false">
      启用一个按钮，让用户通过拉取请求建议修改
    </ResponseField>

    <ResponseField name="raiseIssue" type="boolean" default="false">
      启用一个按钮，让用户针对文档提出问题
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="modeToggle" type="ModeToggle">
  自定义深色模式切换按钮。

  <Expandable title="ModeToggle">
    <ResponseField name="default" type={'"light" or "dark"'}>
      为新用户强制启用浅色或深色模式。若未设置，则默认跟随用户操作系统的偏好设置。
    </ResponseField>

    <ResponseField name="isHidden" type="boolean" default="false">
      设为 true 可隐藏浅色/深色模式切换按钮。你可以将 `isHidden` 与 `default` 搭配使用，强制文档仅使用浅色或深色模式。例如：

      <CodeGroup>
        ```json Only Dark Mode
        "modeToggle": {
          "default": "dark",
          "isHidden": true
        }
        ```

        ```json Only Light Mode
        "modeToggle": {
          "default": "light",
          "isHidden": true
        }
        ```
      </CodeGroup>
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="backgroundImage" type="string">
  显示在每个页面后面的背景图片。示例可参见
  [Infisical](https://infisical.com/docs) 和 [FRPC](https://frpc.io)。
</ResponseField>
